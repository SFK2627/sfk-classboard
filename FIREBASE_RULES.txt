rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function isAdmin() {
      return request.auth != null
        && request.auth.token.email == "admin@sfk-classboard.app";
    }

    function isOfficer() {
      return request.auth != null
        && request.auth.token.email == "officers@sfk-classboard.app";
    }

    function isStaff() {
      return isAdmin() || isOfficer();
    }

    function hasActiveChatProfile() {
      return request.auth != null
        && exists(/databases/$(database)/documents/chatProfiles/$(request.auth.uid))
        && get(/databases/$(database)/documents/chatProfiles/$(request.auth.uid)).data.Active == true
        && get(/databases/$(database)/documents/chatProfiles/$(request.auth.uid)).data.Blocked != true
        && (
          !("MustChangePin" in get(/databases/$(database)/documents/chatProfiles/$(request.auth.uid)).data)
          || get(/databases/$(database)/documents/chatProfiles/$(request.auth.uid)).data.MustChangePin == false
        );
    }

    function isChatMember() {
      return isStaff() || hasActiveChatProfile();
    }

    function isChatUnlocked() {
      return isAdmin()
        || !exists(/databases/$(database)/documents/chatConfig/main)
        || get(/databases/$(database)/documents/chatConfig/main).data.Locked != true;
    }

    function isMediaAllowed() {
      return !exists(/databases/$(database)/documents/chatConfig/main)
        || get(/databases/$(database)/documents/chatConfig/main).data.AllowMedia != false;
    }

    function isWatchPartyAllowed() {
      return !exists(/databases/$(database)/documents/chatConfig/main)
        || get(/databases/$(database)/documents/chatConfig/main).data.WatchPartyEnabled != false;
    }

    function isValidNewWatchRequest(userId) {
      return isChatMember()
        && !isStaff()
        && isWatchPartyAllowed()
        && request.auth.uid == userId
        && request.resource.data.RequesterUID == request.auth.uid
        && request.resource.data.RequesterName is string
        && request.resource.data.Title is string
        && request.resource.data.OriginalURL is string
        && request.resource.data.VideoID is string
        && request.resource.data.Provider in [
          "youtube", "tiktok", "instagram", "drive", "bilibili",
          "vimeo", "direct", "hls", "dash"
        ]
        && request.resource.data.Status == "pending"
        && request.resource.data.keys().hasOnly([
          "RequesterUID", "RequesterName", "Title", "Provider", "VideoID",
          "InstagramType", "OriginalURL", "Status", "CreatedAt", "UpdatedAt"
        ]);
    }

    function capsuleIsUnlocked() {
      return exists(/databases/$(database)/documents/timeCapsuleSettings/main)
        && get(/databases/$(database)/documents/timeCapsuleSettings/main).data.UnlockAt <= request.time;
    }

    function capsuleSubmissionsOpen() {
      return exists(/databases/$(database)/documents/timeCapsuleSettings/main)
        && get(/databases/$(database)/documents/timeCapsuleSettings/main).data.AllowSubmissions == true
        && get(/databases/$(database)/documents/timeCapsuleSettings/main).data.SubmissionDeadline >= request.time;
    }

    function validCapsuleImageURL() {
      return request.resource.data.ImageURL is string
        && request.resource.data.ImageURL.size() <= 560000
        && (
          request.resource.data.ImageURL.size() == 0
          || request.resource.data.ImageURL.matches("^https://.*")
          || request.resource.data.ImageURL.matches("^sfk-media://capsule/[A-Za-z0-9_-]{1,240}$")
          || request.resource.data.ImageURL.matches("^data:image/(jpeg|jpg|png|webp);base64,[A-Za-z0-9+/=]+$")
        );
    }

    function validCapsuleMedia() {
      return request.auth != null
        && request.resource.data.OwnerUID == request.auth.uid
        && request.resource.data.OwnerName is string
        && request.resource.data.OwnerName.size() > 0
        && request.resource.data.OwnerName.size() <= 120
        && request.resource.data.OwnerRole in ["student", "officer", "admin"]
        && request.resource.data.EntryID is string
        && request.resource.data.EntryID.size() > 0
        && request.resource.data.EntryID.size() <= 200
        && request.resource.data.Name is string
        && request.resource.data.Name.size() > 0
        && request.resource.data.Name.size() <= 180
        && request.resource.data.MimeType == "image/jpeg"
        && request.resource.data.Data is string
        && request.resource.data.Data.size() > 0
        && request.resource.data.Data.size() <= 850000
        && request.resource.data.BytesApprox is number
        && request.resource.data.BytesApprox > 0
        && request.resource.data.BytesApprox <= 700000
        && request.resource.data.CreatedAt == request.time
        && request.resource.data.UpdatedAt == request.time
        && request.resource.data.keys().hasOnly([
          "EntryID", "OwnerUID", "OwnerName", "OwnerRole",
          "Name", "MimeType", "Data", "BytesApprox",
          "CreatedAt", "UpdatedAt"
        ]);
    }

    function capsuleMediaEntryIsApproved(entryId) {
      return entryId is string
        && exists(/databases/$(database)/documents/timeCapsuleEntries/$(entryId))
        && get(/databases/$(database)/documents/timeCapsuleEntries/$(entryId)).data.Status == "approved";
    }

    function validCapsuleEntry() {
      return request.resource.data.AuthorUID == request.auth.uid
        && request.resource.data.AuthorName is string
        && request.resource.data.AuthorName.size() > 0
        && request.resource.data.AuthorName.size() <= 120
        && request.resource.data.AuthorRole in ["student", "officer", "admin"]
        && request.resource.data.Type in ["memory", "message", "goal", "prediction", "photo"]
        && request.resource.data.Text is string
        && request.resource.data.Text.size() <= 1200
        && validCapsuleImageURL()
        && (
          request.resource.data.Text.size() > 0
          || request.resource.data.ImageURL.size() > 0
        );
    }

    function isAllowedProfileColor(color) {
      return color is string
        && color.matches("^#[0-9A-Fa-f]{6}$");
    }

    function isNotMuted() {
      return isStaff()
        || !("MutedUntil" in get(/databases/$(database)/documents/chatProfiles/$(request.auth.uid)).data)
        || get(/databases/$(database)/documents/chatProfiles/$(request.auth.uid)).data.MutedUntil <= request.time;
    }

    function isHeartLedgerDocument(docId) {
      return docId.matches("^heartV3_.*")
        && request.resource.data.Kind == "ClassBoardHeartLedgerV3"
        && request.resource.data.TargetType in ["announcement", "memory"]
        && request.resource.data.TargetKey is string
        && request.resource.data.DeviceID is string
        && request.resource.data.Active is bool
        && request.resource.data.keys().hasOnly([
          "Kind", "TargetType", "TargetKey", "DeviceID", "Active",
          "UpdatedAtText", "UpdatedAt"
        ]);
    }

    match /settings/{docId} {
      allow read: if true;
      allow create, update: if isStaff() || isHeartLedgerDocument(docId);
      allow delete: if isAdmin() || docId.matches("^heartV3_.*");
    }

    match /schedule/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /announcements/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /announcementMedia/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /thingsToBring/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /adviserReminders/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /prayerLeaders/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /dailyQuotes/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /birthdays/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /tickerMessages/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /dailyInfo/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /memories/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /memoryMedia/{docId} {
      allow read: if true;
      allow create, update: if isStaff();
      allow delete: if isAdmin();
    }

    match /chatProfiles/{userId} {
      allow get: if isAdmin() || (request.auth != null && request.auth.uid == userId);
      allow list, create, delete: if isAdmin();
      allow update: if isAdmin()
        || (
          request.auth != null
          && request.auth.uid == userId
          && resource.data.MustChangePin == true
          && request.resource.data.MustChangePin == false
          && isAllowedProfileColor(request.resource.data.AvatarColor)
          && request.resource.data.ColorChangedAt == request.time
          && request.resource.data.diff(resource.data).affectedKeys()
            .hasOnly(["MustChangePin", "AvatarColor", "ColorChangedAt", "UpdatedAt"])
        )
        || (
          request.auth != null
          && request.auth.uid == userId
          && resource.data.MustChangePin == false
          && request.resource.data.MustChangePin == false
          && isAllowedProfileColor(request.resource.data.AvatarColor)
          && request.resource.data.ColorChangedAt == request.time
          && (
            !("ColorChangedAt" in resource.data)
            || resource.data.ColorChangedAt <= request.time - duration.value(3, "d")
          )
          && request.resource.data.diff(resource.data).affectedKeys()
            .hasOnly(["AvatarColor", "ColorChangedAt", "UpdatedAt"])
        );
    }

    match /chatMessages/{messageId} {
      allow read: if isChatMember();
      allow create: if isChatMember()
        && isChatUnlocked()
        && isNotMuted()
        && request.resource.data.SenderUID == request.auth.uid
        && request.resource.data.SenderName is string
        && request.resource.data.Text is string
        && request.resource.data.Text.size() > 0
        && request.resource.data.Text.size() <= 500
        && request.resource.data.Removed == false
        && (("MediaURL" in request.resource.data) == ("MediaType" in request.resource.data))
        && (
          !("MediaURL" in request.resource.data)
          || (
            isMediaAllowed()
            && request.resource.data.MediaURL is string
            && request.resource.data.MediaURL.size() > 0
            && request.resource.data.MediaURL.size() <= 2000
            && request.resource.data.MediaURL.matches("^https?://.*")
            && request.resource.data.MediaType in ["image", "video", "audio", "youtube"]
          )
        )
        && (
          (
            !("Type" in request.resource.data)
            && request.resource.data.keys().hasOnly([
              "Text", "SenderUID", "SenderName", "SenderRole", "Removed", "CreatedAt",
              "ReplyToID", "ReplyToName", "ReplyToText", "MediaURL", "MediaType"
            ])
          )
          || (
            request.resource.data.Type == "poll"
            && request.resource.data.PollOptions is list
            && request.resource.data.PollOptions.size() >= 2
            && request.resource.data.PollOptions.size() <= 12
            && request.resource.data.keys().hasOnly([
              "Type", "Text", "PollOptions", "SenderUID", "SenderName",
              "SenderRole", "Removed", "CreatedAt", "PollMultiple",
              "PollAnonymous", "PollClosed", "PollEndsAt"
            ])
          )
        );
      allow update: if isAdmin()
        || (
          isChatMember()
          && resource.data.SenderUID == request.auth.uid
          && request.time < resource.data.CreatedAt + duration.value(5, "m")
          && request.resource.data.SenderUID == resource.data.SenderUID
          && (
            (
              request.resource.data.Removed == true
              && request.resource.data.Text == ""
              && request.resource.data.diff(resource.data).affectedKeys()
                .hasOnly(["Text", "Removed", "RemovedBy", "RemovedAt"])
            )
            || (
              isChatUnlocked()
              && !("Type" in resource.data)
              && resource.data.Removed == false
              && request.resource.data.Removed == false
              && request.resource.data.Text is string
              && request.resource.data.Text.size() > 0
              && request.resource.data.Text.size() <= 500
              && request.resource.data.Edited == true
              && request.resource.data.diff(resource.data).affectedKeys()
                .hasOnly(["Text", "Edited", "EditedAt"])
            )
          )
        )
        || (
          isChatMember()
          && resource.data.Type == "poll"
          && resource.data.SenderUID == request.auth.uid
          && resource.data.PollClosed == false
          && request.resource.data.PollClosed == true
          && request.resource.data.diff(resource.data).affectedKeys()
            .hasOnly(["PollClosed", "PollClosedAt"])
        );
      allow delete: if isAdmin();

      match /reactions/{userId} {
        allow read: if isChatMember();
        allow create, update: if isChatMember()
          && request.auth.uid == userId
          && get(/databases/$(database)/documents/chatMessages/$(messageId)).data.Removed == false
          && request.resource.data.UID == request.auth.uid
          && request.resource.data.Emoji in ["🫶", "👍", "❤️", "😂", "😮", "😢", "🙏", "✅"]
          && request.resource.data.keys().hasOnly(["Emoji", "UID", "Name", "UpdatedAt"]);
        allow delete: if isAdmin() || (isChatMember() && request.auth.uid == userId);
      }

      match /votes/{userId} {
        allow read: if isChatMember();
        allow create, update: if isChatMember()
          && request.auth.uid == userId
          && get(/databases/$(database)/documents/chatMessages/$(messageId)).data.Removed == false
          && get(/databases/$(database)/documents/chatMessages/$(messageId)).data.Type == "poll"
          && get(/databases/$(database)/documents/chatMessages/$(messageId)).data.PollClosed == false
          && (
            !("PollEndsAt" in get(/databases/$(database)/documents/chatMessages/$(messageId)).data)
            || get(/databases/$(database)/documents/chatMessages/$(messageId)).data.PollEndsAt > request.time
          )
          && request.resource.data.UID == request.auth.uid
          && request.resource.data.OptionIndexes is list
          && request.resource.data.OptionIndexes.size() >= 1
          && request.resource.data.OptionIndexes.size()
            <= get(/databases/$(database)/documents/chatMessages/$(messageId)).data.PollOptions.size()
          && request.resource.data.keys().hasOnly(["UID", "Name", "OptionIndexes", "UpdatedAt"]);
        allow delete: if isAdmin() || (isChatMember() && request.auth.uid == userId);
      }
    }

    match /chatMeta/{docId} {
      allow read: if docId == "main";
      allow create: if docId == "main"
        && isChatMember()
        && request.resource.data.TotalMessages == 1
        && request.resource.data.keys().hasOnly(["TotalMessages", "LatestAt"]);
      allow update: if docId == "main"
        && (
          isAdmin()
          || (
            isChatMember()
            && request.resource.data.TotalMessages == resource.data.TotalMessages + 1
            && request.resource.data.diff(resource.data).affectedKeys()
              .hasOnly(["TotalMessages", "LatestAt"])
          )
        );
      allow delete: if isAdmin();
    }

    match /chatConfig/{docId} {
      allow read: if isChatMember();
      allow create, update, delete: if isAdmin();
    }

    match /chatDirectory/{userId} {
      allow read: if isChatMember();
      allow create, update: if isAdmin()
        || (
          request.auth != null
          && request.auth.uid == userId
          && getAfter(/databases/$(database)/documents/chatProfiles/$(userId)).data.Active == true
          && getAfter(/databases/$(database)/documents/chatProfiles/$(userId)).data.Blocked != true
          && getAfter(/databases/$(database)/documents/chatProfiles/$(userId)).data.MustChangePin == false
          && request.resource.data.Name == getAfter(/databases/$(database)/documents/chatProfiles/$(userId)).data.Name
          && request.resource.data.Role == "student"
          && request.resource.data.AvatarColor == getAfter(/databases/$(database)/documents/chatProfiles/$(userId)).data.AvatarColor
          && isAllowedProfileColor(request.resource.data.AvatarColor)
          && request.resource.data.keys().hasOnly(["Name", "Role", "AvatarColor", "UpdatedAt"])
        );
      allow delete: if isAdmin();
    }

    match /chatSaved/{userId}/items/{messageId} {
      allow read: if isAdmin() || (isChatMember() && request.auth.uid == userId);
      allow create, update: if isChatMember() && request.auth.uid == userId;
      allow delete: if isAdmin() || (isChatMember() && request.auth.uid == userId);
    }

    match /chatReports/{reportId} {
      allow create: if isChatMember()
        && request.resource.data.ReporterUID == request.auth.uid;
      allow read, update, delete: if isAdmin();
    }

    match /chatWatchRequests/{userId} {
      allow read: if isStaff()
        || (isChatMember() && request.auth.uid == userId);
      allow create: if isValidNewWatchRequest(userId);
      allow update: if isStaff()
        || (resource.data.Status in ["approved", "declined"]
          && isValidNewWatchRequest(userId));
      allow delete: if isStaff()
        || (isChatMember() && request.auth.uid == userId);
    }

    match /chatWatchParty/{docId} {
      allow read: if isChatMember();
      allow create: if isStaff() && docId == "main";
      allow update: if isStaff()
        || (
          docId == "main"
          && isChatMember()
          && resource.data.Active == true
          && resource.data.HostRole == "student"
          && resource.data.HostUID == request.auth.uid
          && resource.data.HostExpiresAt > request.time
          && request.resource.data.HostUID == resource.data.HostUID
          && request.resource.data.VideoID == resource.data.VideoID
          && request.resource.data.Provider == resource.data.Provider
          && request.resource.data.OriginalURL == resource.data.OriginalURL
          && request.resource.data.PlaybackState in ["playing", "paused"]
          && request.resource.data.Position is number
          && request.resource.data.Position >= 0
          && (
            !("VolumeSyncEnabled" in request.resource.data)
            || request.resource.data.VolumeSyncEnabled is bool
          )
          && (
            !("VolumeLevel" in request.resource.data)
            || (
              request.resource.data.VolumeLevel is number
              && request.resource.data.VolumeLevel >= 0
              && request.resource.data.VolumeLevel <= 1
            )
          )
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
            "Active", "PlaybackState", "Position", "StateUpdatedAt",
            "VolumeSyncEnabled", "VolumeLevel",
            "UpdatedAt", "EndedBy", "EndedAt"
          ])
        );
      allow delete: if isStaff();
    }

    match /chatWatchViewers/{userId} {
      allow read: if isChatMember();
      allow create, update: if isChatMember()
        && request.auth.uid == userId
        && request.resource.data.UID == request.auth.uid
        && request.resource.data.Name is string
        && request.resource.data.PartySessionID is string
        && request.resource.data.ExpiresAt is timestamp
        && request.resource.data.UpdatedAt is timestamp
        && request.resource.data.keys().hasOnly([
          "UID", "Name", "PartySessionID", "ExpiresAt", "UpdatedAt"
        ]);
      allow delete: if isAdmin()
        || (isChatMember() && request.auth.uid == userId);
    }

    match /chatScheduled/{messageId} {
      allow read: if isAdmin() || (isChatMember() && resource.data.PublishAt <= request.time);
      allow create, update, delete: if isAdmin();
    }

    match /chatTyping/{userId} {
      allow read: if isChatMember();
      allow create, update: if isChatMember()
        && request.auth.uid == userId
        && request.resource.data.UID == request.auth.uid
        && request.resource.data.keys().hasOnly(["UID", "Name", "ExpiresAt"]);
      allow delete: if isAdmin() || (isChatMember() && request.auth.uid == userId);
    }

    match /chatReadReceipts/{userId} {
      allow read: if isChatMember();
      allow create, update: if isChatMember()
        && request.auth.uid == userId
        && request.resource.data.UID == request.auth.uid
        && request.resource.data.keys().hasOnly(["UID", "Name", "LastSeenAt"]);
      allow delete: if isAdmin();
    }

    match /chatAuditLogs/{logId} {
      allow read: if isAdmin();
      allow create: if isChatMember()
        && request.resource.data.ActorUID == request.auth.uid
        && request.resource.data.ActorName is string
        && request.resource.data.ActorName.size() > 0
        && request.resource.data.ActorName.size() <= 120
        && request.resource.data.ActorRole in ["student", "officer", "admin"]
        && request.resource.data.Event in [
          "MESSAGE_SENT", "POLL_SENT", "MEDIA_SENT", "MESSAGE_EDITED",
          "MESSAGE_DELETED", "REACTION_ADDED", "REACTION_CHANGED",
          "REACTION_REMOVED", "MESSAGE_PINNED", "MESSAGE_UNPINNED",
          "PRIORITY_CHANGED", "POLL_CLOSED", "USER_MUTED",
          "REPORT_CREATED", "WATCH_PARTY_STARTED", "WATCH_PARTY_ENDED",
          "CLASS_THEME_CHANGED"
        ]
        && request.resource.data.CreatedAt == request.time
        && request.resource.data.keys().hasOnly([
          "Event", "ActorUID", "ActorName", "ActorRole", "MessageID",
          "TargetUID", "TargetName", "Text", "PreviousText", "NewText",
          "Emoji", "PreviousEmoji", "MediaType", "Details", "CreatedAt"
        ]);
      allow update, delete: if false;
    }

    match /chatModerationLogs/{logId} {
      allow read, create: if isAdmin();
      allow update, delete: if false;
    }

    match /timeCapsuleSettings/{docId} {
      allow read: if isChatMember();
      allow create, update: if isAdmin() && docId == "main";
      allow delete: if false;
    }

    match /timeCapsulePublic/{docId} {
      allow read: if docId == "main";
      allow create, update: if isAdmin() && docId == "main";
      allow delete: if false;
    }

    match /timeCapsuleMedia/{mediaId} {
      allow read: if isAdmin()
        || (isChatMember() && resource.data.OwnerUID == request.auth.uid)
        || (
          isChatMember()
          && capsuleIsUnlocked()
          && capsuleMediaEntryIsApproved(resource.data.EntryID)
        );
      allow create: if isChatMember() && (isAdmin() || capsuleSubmissionsOpen()) && validCapsuleMedia();
      allow update: if false;
      allow delete: if isAdmin()
        || (isChatMember() && resource.data.OwnerUID == request.auth.uid);
    }

    match /timeCapsuleEntries/{entryId} {
      allow read: if isAdmin()
        || (isChatMember() && resource.data.AuthorUID == request.auth.uid)
        || (
          isChatMember()
          && capsuleIsUnlocked()
          && resource.data.Status == "approved"
        );
      allow create: if isChatMember()
        && validCapsuleEntry()
        && (
          (isAdmin() && request.resource.data.Status == "approved")
          || (!isAdmin() && request.resource.data.Status == "pending")
        )
        && request.resource.data.CreatedAt == request.time
        && request.resource.data.UpdatedAt == request.time
        && request.resource.data.keys().hasOnly([
          "AuthorUID", "AuthorName", "AuthorRole", "Type", "Text",
          "ImageURL", "Status", "CreatedAt", "UpdatedAt"
        ])
        && (isAdmin() || capsuleSubmissionsOpen());
      allow update: if isAdmin()
        || (
          isChatMember()
          && capsuleSubmissionsOpen()
          && resource.data.AuthorUID == request.auth.uid
          && resource.data.Status == "pending"
          && request.resource.data.AuthorUID == resource.data.AuthorUID
          && request.resource.data.AuthorName == resource.data.AuthorName
          && request.resource.data.AuthorRole == resource.data.AuthorRole
          && request.resource.data.Status == "pending"
          && validCapsuleEntry()
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
            "Type", "Text", "ImageURL", "UpdatedAt"
          ])
        );
      allow delete: if isAdmin()
        || (
          isChatMember()
          && capsuleSubmissionsOpen()
          && resource.data.AuthorUID == request.auth.uid
          && resource.data.Status == "pending"
        );
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
